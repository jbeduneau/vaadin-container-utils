11   * the beanClass

13   * @author Vincent Demeester <vincent@demeester.fr>
14   * 
15   */
16  public class AttributeReaderAlgorithm implements PropertyReaderAlgorithm {
17  
18  	/**
19  	 * Constant for default valuer of withSuper attribute.
20  	 */
21  	private static final boolean DEFAULT_WITHSUPER = true;
22  	private static final List<String> DEFAULT_IGNORED = Arrays
23  			.asList(new String[] { "serialVersionUID" });
24

26  	 * List of attribute to ignore.
27  	 */
28  	private final Collection<String> ignoredAttributes;
29  	private final boolean withSuper;
30  
31  	/**
32  	 * Create an {@link AttributeReaderAlgorithm} with default ignored
33  	 * attributes ("serialVersionUID") and with <code>super</code> class lookup.
34  	 */
35  	public AttributeReaderAlgorithm() {
36  		this(DEFAULT_IGNORED, DEFAULT_WITHSUPER);
37  	}

39  	/**
40  	 * Create an {@link AttributeReaderAlgorithm} with default ignored
41  	 * attributes ("serialVersionUID").
42  	 * 
43  	 * @param withSuper
44  	 *            look up for <code>super</class> class.
45  	 */
46  	public AttributeReaderAlgorithm(boolean withSuper) {
47  		this(DEFAULT_IGNORED, withSuper);
48  	}
49  
50  	/**
51  	 * Create an {@link AttributeReaderAlgorithm} with <code>super</code> class
52  	 * lookup.
53  	 * 
54  	 * @param ignoredAttributes
55  	 *            list of attributes to ignore.
56  	 */
57  	public AttributeReaderAlgorithm(Collection<String> ignoredAttributes) {
58  		this(ignoredAttributes, DEFAULT_WITHSUPER);
59  	}
60  
61  	/**
62  	 * Create an {@link AttributeReaderAlgorithm}.
63  	 * 
64  	 * @param ignoredAttributes
65  	 *            list of attributes to ignore.
66  	 * @param withSuper
67  	 *            look up for <code>super</class> class.
68  	 */
69  	public AttributeReaderAlgorithm(Collection<String> ignoredAttributes,
70  			boolean withSuper) {
71  		this.ignoredAttributes = ignoredAttributes;
72  		this.withSuper = withSuper;
73  	}
74  
75  	/**
76  	 * {@inheritDoc}
77  	 * 
78  	 * If the ignoredAttributes attribute is null, we won't fail, it will be
79  	 * just as if it is an empty List.
80  	 * 
81  	 * @throw {@link IllegalArgumentException} if beanClass is null.
82  	 */
83  	@Override
84  	public List<PropertyMetadata> getProperties(Class<?> beanClass) {
85  		if (beanClass == null) {
86  			throw new IllegalArgumentException("beanClass cannot be null.");
87  		}
88  		List<PropertyMetadata> metadatas = new LinkedList<PropertyMetadata>();
89  		List<Field> fields = new LinkedList<Field>();
90  		fields.addAll(Arrays.asList(beanClass.getDeclaredFields()));
91  		if (withSuper) {
92  			Class<?> superClass = beanClass.getSuperclass();
93  			while (superClass != Object.class) {
94  				fields.addAll(Arrays.asList(superClass.getDeclaredFields()));
95  				superClass = superClass.getSuperclass();
96  			}
97  		}
98  		for (Field field : fields) {
99  			if (ignoredAttributes == null
100 					|| !ignoredAttributes.contains(field.getName())) {
101 				String propertyName = field.getName();
102 				Class<?> propertyClass = field.getType();
103 				String propertyAttribute = field.getName();
104 				PropertyMetadata metadata = new PropertyMetadata(propertyName,
105 						propertyClass, null, propertyAttribute);
106 				metadatas.add(metadata);
107 			}
108 		}
109 		return metadatas;
110 	}
