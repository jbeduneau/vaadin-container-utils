25  	/**
26  	 * Initialize a container with the given type.
27  	 * 
28  	 * @param containerClass
29  	 *            type of the container
30  	 * @return a container.
31  	 * @throws IllegalAccessException
32  	 * @throws InstantiationException
33  	 */
34  	public static Container initContainer(
35  			Class<? extends Container> containerClass)
36  			throws InstantiationException, IllegalAccessException {
37  		Container container = null;
38  		if (containerClass.isInterface()) {
39  			if (containerClass == Filterable.class) {
40  				// Default Filterable is Indexed
41  				container = new IndexedContainer();
42  			} else if (containerClass == Hierarchical.class) {
43  				// Default Hierarchical is HierarchicalContainer
44  				container = new HierarchicalContainer();
45  			} else {
46  				throw new InstantiationException(containerClass
47  						+ " not supported.");
48  			}
49  		} else {
50  			container = containerClass.newInstance();
51  		}
52  		return container;
53  	}
