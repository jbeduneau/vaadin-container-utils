48  	/**
49  	 * Get the type of the given propertyAttribute
50  	 * 
51  	 * @param propertyAttribute
52  	 * @return
53  	 * @throws SecurityException
54  	 * @throws NoSuchFieldException
55  	 */
56  	protected Class<?> getPropertyType(String propertyAttribute)
57  			throws SecurityException, NoSuchFieldException {
58  		Class<?> ret = null;
59  		if (propertyAttribute.contains(".")) {
60  			String fieldName = propertyAttribute.split("\\.")[0];
61  			String subFieldName = propertyAttribute.split("\\.")[1];
62  			Class<?> fieldClass = getOriginalEntityClass().getDeclaredField(
63  					fieldName).getType();
64  			ret = fieldClass.getDeclaredField(subFieldName).getType();
65  		} else {
66  			ret = getOriginalEntityClass().getDeclaredField(propertyAttribute)
67  					.getType();
68  		}
69  		return ret;
70  	}

72  	/**
73  	 * Get the annotated type by walking in superclass to find the annotation
74  	 * 
75  	 * @param entityClass
76  	 *            The type that should be annotated
77  	 * @param annotationClass
78  	 *            the type of the annotation
79  	 * @return the "real" type annotated, either entityClass or a parent.
80  	 */
81  	public static Class<?> getAnnotatedClass(Class<?> entityClass,
82  			Class<? extends Annotation> annotationClass) {
83  		Class<?> ret = null;
84  		if (!entityClass.isAnnotationPresent(annotationClass)) {
85  			// On remonte dans la hierarchie jusqu'à Object
86  			if (entityClass.getSuperclass() != Object.class) {
87  				ret = getAnnotatedClass(entityClass.getSuperclass(),
88  						annotationClass);
89  			}
90  		} else {
91  			ret = entityClass;
92  		}
93  		return ret;
94  	}
